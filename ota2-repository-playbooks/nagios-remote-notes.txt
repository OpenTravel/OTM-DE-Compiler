Instrumentation of Remote Server for Nagios Monitoring
------------------------------------------------------
Centos 7
---
yum install -y gcc glibc glibc-common make gettext automake autoconf wget openssl-devel net-snmp net-snmp-utils epel-release
yum install -y perl-Net-SNMP
cd /tmp
wget -O http://nagios-plugins.org/download/nagios-plugins-2.3.1.tar.gz
tar zxf nagios-plugins-2.3.1.tar.gz
cd /tmp/nagios-plugins-release-2.2.1/
./tools/setup
./configure
make
make install

# NRPE Installation
cd /tmp
wget https://sourceforge.net/projects/nagios/files/nrpe-3.x/nrpe-3.2.1.tar.gz
tar zxf nrpe-3.2.1.tar.gz
cd /tmp/nrpe-3.2.1/
./configure --enable-command-args
make all
make install
make install-plugin
make install-daemon
make install-config
make install-inetd
make install-groups-users
make install-init
echo 'nrpe    5666/tcp' >> /etc/services
systemctl enable nrpe
systemctl start nrpe
*****  Open firewall port 5666/tcp
